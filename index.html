Pretty Code


class Dradis
  include Eventful

  event :new_contact
end

class ConsoleListener
  def on_new_contact(direction, range)
    puts "DRADIS contact! #{range} kilometers"
  end
end

dradis = Dradis.new
dradis.add_listener(ConsoleListener.new)
dradis.on_new_contact 100, 20
